        function updateMargin() {
            const val = document.getElementById('marginSlider').value;
            document.getElementById('marginBadge').innerText = val + '%';
            document.getElementById('footer-margin').innerText = val + '%';
            document.getElementById('s-mkpct').innerText = val;
            recalc();
        }

        function recalc() {
            let matTotal = 0;
            document.querySelectorAll('#materialBody tr').forEach(row => {
                const qtyInput = row.querySelector('.mat-qty');
                const costInput = row.querySelector('.mat-cost');
                if (!qtyInput || !costInput) return;
                const qty = parseFloat(qtyInput.value) || 0;
                const cost = parseFloat(costInput.value) || 0;
                const ext = qty * cost;
                row.querySelector('.mat-ext').innerText = '$' + ext.toFixed(2);
                matTotal += ext;
            });
            document.getElementById('matTotal').innerText = '$' + matTotal.toFixed(2);

            let laborTotal = 0;
            document.querySelectorAll('#laborBody tr').forEach(row => {
                const hoursInput = row.querySelector('.lab-hours');
                const rateInput = row.querySelector('.lab-rate');
                if (!hoursInput || !rateInput) return;
                const hours = parseFloat(hoursInput.value) || 0;
                const rate = parseFloat(rateInput.value) || 0;
                const ext = hours * rate;
                row.querySelector('.lab-ext').innerText = '$' + ext.toFixed(2);
                laborTotal += ext;
            });
            document.getElementById('laborTotal').innerText = '$' + laborTotal.toFixed(2);

            const refLbs = parseFloat(document.getElementById('refLbs').value) || 0;
            const refRate = parseFloat(document.getElementById('refRate').value) || 0;
            const refTotal = refLbs * refRate;

            const permit = parseFloat(document.getElementById('permitFee').value) || 0;
            const disposal = parseFloat(document.getElementById('disposalFee').value) || 0;
            const rawCost = matTotal + laborTotal + refTotal + permit + disposal;

            const marginPct = parseFloat(document.getElementById('marginSlider').value) || 0;
            const marginMultiplier = (marginPct >= 100) ? 100 : (1 - (marginPct / 100));
            const sellPriceBeforeDisc = rawCost / marginMultiplier;
            const markup = sellPriceBeforeDisc - rawCost;

            const discount = parseFloat(document.getElementById('discount').value) || 0;
            const finPct = parseFloat(document.getElementById('financing').value) || 0;
            const finFee = (sellPriceBeforeDisc - discount) * (finPct / 100);

            const taxRate = parseFloat(document.getElementById('taxRate').value) || 0;
            const taxable = Math.max(0, sellPriceBeforeDisc - discount + finFee);
            const tax = taxable * (taxRate / 100);

            const totalBid = taxable + tax;

            // Summary
            document.getElementById('s-mat').innerText = '$' + matTotal.toFixed(2);
            document.getElementById('s-lab').innerText = '$' + laborTotal.toFixed(2);
            document.getElementById('s-ref').innerText = '$' + refTotal.toFixed(2);
            document.getElementById('s-fees').innerText = '$' + (permit + disposal).toFixed(2);
            document.getElementById('s-raw').innerText = '$' + rawCost.toFixed(2);
            document.getElementById('s-markup').innerText = '$' + markup.toFixed(2);
            document.getElementById('s-disc').innerText = '-$' + discount.toFixed(2);
            document.getElementById('s-fin').innerText = '$' + finFee.toFixed(2);
            document.getElementById('s-tax').innerText = '$' + tax.toFixed(2);
            document.getElementById('s-total').innerText = '$' + totalBid.toFixed(2);
            document.getElementById('footer-total').innerText = '$' + totalBid.toFixed(2);

            // Tiers
            document.getElementById('t-good').innerHTML = '$' + totalBid.toFixed(2) + '<small>Standard Efficiency</small>';
            document.getElementById('t-better').innerHTML = '$' + (totalBid * 1.08).toFixed(2) + '<small>Mid Efficiency (+8%)</small>';
            document.getElementById('t-best').innerHTML = '$' + (totalBid * 1.18).toFixed(2) + '<small>High Efficiency (+18%)</small>';

            // Sticky Bar
            if (window.updateMintStickyBar) {
                window.updateMintStickyBar({
                    'Bid Total': '$' + totalBid.toFixed(0),
                    'Profit': '$' + markup.toFixed(0)
                }, '/hvac-bid-calculator/');
            }
        }
